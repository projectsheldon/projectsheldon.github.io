POST /sheldon/resell/generate

Get a random license key from your reseller pool.

Request Body

{
  "token": "api_key"
}

Response

{
  "ok": true,
  "key": "XXXXXX-XXXXXXX-XX",
  "expiresAt": 1735689600000
}
import requests

url = "BASE_URL/sheldon/resell/generate"
data = {"token": "api_key"}

response = requests.post(url, json=data)
result = response.json()

if result.get("ok"):
    print(f"License Key: {result['key']}")
    print(f"Expires: {result['expiresAt']}")
else:
    print(f"Error: {result.get('error')}")
const url = "BASE_URL/sheldon/resell/generate";

fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ token: "api_key" })
})
.then(res => res.json())
.then(result => {
    if (result.ok) {
        console.log(`License Key: ${result.key}`);
        console.log(`Expires: ${result.expiresAt}`);
    } else {
        console.log(`Error: ${result.error}`);
    }
});
POST /sheldon/resell/assign

Generate and assign a license to a specific Discord user. The license will be linked to their Discord ID.

Request Body

{
  "token": "api_key",
  "discordId": "123456789",
  "license": "XXXXXX-XXXXXXX-XX"
}

Response

{
  "ok": true,
  "key": "XXXXXX-XXXXXXX-XX",
  "expiresAt": 1735689600000
}
import requests

url = "BASE_URL/sheldon/resell/assign"
data = {
    "token": "api_key",
    "discordId": "123456789",
}

response = requests.post(url, json=data)
result = response.json()

if result.get("ok"):
    print(f"License Key: {result['key']}")
    print(f"Expires: {result['expiresAt']}")
else:
    print(f"Error: {result.get('error')}")
const url = "BASE_URL/sheldon/resell/assign";
